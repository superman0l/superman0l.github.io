2. 对一个数组可以对数组的每个数都+k或-k（无限次），要将数组中的每个数都操作至相等，这个k最大是多少

   求各个数差值的最大公因数

   ```c++
   int res = 0, m = 0;
   for(int i = 0;i<n;i++){
   	m = max(m, a[i]);
   }
   res = m - a[0];
   for(int i = 1;i<n;i++){
   	res = gcd(res, m - a[i]);
   }
   ```

3. 有一个迷宫，给定一个终点，你可以等概率出现在迷宫的各个可以到达终点的位置（也有概率直接出现在终点），在你出现在迷宫的某个位置后可以有一条最短路径到达终点，问到达终点的期望路径，结果输出最简分数`p/q`

   从终点开始bfs搜索所有点即可

   ```c++
   struct node{
   	int x,y;
       int pace;
   }
   
   vector<vector<int>> dir = {{0,1}, {1,0}, {0,-1}, {-1,0}};
       
   queue<node> q;
   vector<vector<int>> visited(n,vector<int>(m,0));
   node a;
   a.x = endx;a.y=endy;a.pace = 0;
   q.push(a);
   
   int respace = 0, resn = 0;
   while(!q.empty()){
       node t = q.top();
       respace += t.pace;
       resn ++;
       visited[t.x][t.y]=1;
       q.pop();
       for(int i = 0;i<4;i++){
           int tx = t.x + dir[i][0];
           int ty = t.y + dir[i][1];
           if(tx>=0 && tx<n && ty>=0 && ty<m && !visited[tx][ty]){
               node t1 = t;t1.x = tx;t1.y = ty;
               t1.pace++;
               visited[tx][ty]=1;
               q.push(t1);
           }
       }
       
       int pn = gcd(respace, resn);
       cout<< respace / pn << " " << resn / pn;
   }
   ```

4. 给一个n*m的地图，里面有k个机器人，有q个操作，每次可以操作机器人上下左右移动，当移动到面前有机器人或移动到边界时停止，问每次操作之后机器人的位置

   ```c++
   //懒得写了 用丑陋的ac代码将就一下
   //给的位置都是1 ~ n和1 ~ m，个人习惯取0 ~ n-1，所以读了之后--，输出的时候++
   
   int main() {
       int n, m; cin >> n >> m;
       int k; cin >> k;
       vector<pair<int, int>> r(k);
       int q; cin >> q;
   
       map<int, set<int>> xr, yr;
   
       for (int i = 0; i < k; i++) {
           int x, y; cin >> x >> y;
           x--; y--;
           r[i].first = x;
           r[i].second = y;
           xr[x].insert(i);
           yr[y].insert(i);
       }
   
       while (q--) {
           int v; cin >> v; v--;
           char c; cin >> c;
           int x = r[v].first;
           int y = r[v].second;
   
           bool haschange = 0;
           if (c == 'L' || c=='R') {
               for (int i : xr[r[v].first]) {
                   if (i != v) {
                       if (c == 'R' && r[i].second - 1 >= r[v].second) {
                           if (!haschange) {
                               y = r[i].second - 1;
                               haschange = 1;
                           }
                           else if (y > r[i].second - 1) y = r[i].second - 1;
                       }
                       else if (c == 'L' && r[i].second + 1 <= r[v].second) {
                           if (!haschange) {
                               y = r[i].second + 1;
                               haschange = 1;
                           }
                           else if (y < r[i].second + 1) y = r[i].second + 1;
                       }
                   }
               }
           }
           else if (c == 'U' || c == 'D') {
               for (int i : yr[r[v].second]) {
                   if (i != v) {
                       if (c == 'U' && r[i].first + 1 <= r[v].first) {
                           if (!haschange) {
                               x = r[i].first + 1;
                               haschange = 1;
                           }
                           else if (x < r[i].first + 1) x = r[i].first + 1;
                       }
                       else if (c == 'D' && r[i].first - 1 >= r[v].first) {
                           if (!haschange) {
                               x = r[i].first - 1;
                               haschange = 1;
                           }
                           else if (x > r[i].first - 1) x = r[i].first - 1;
                       }
                   }
               }
           }
   
           if(!haschange){
               if (c == 'U' || c == 'D') {
                   xr[r[v].first].erase(v);
               }
               else if (c == 'L' || c == 'R') {
                   yr[r[v].second].erase(v);
               }
   
               if (c == 'U') {
                   r[v].first = 0;
               }
               else if (c == 'D') {
                   r[v].first = n - 1;
               }
               else if (c == 'L') {
                   r[v].second = 0;
               }
               else if (c == 'R') {
                   r[v].second = m - 1;
               }
           }
           else {
               xr[r[v].first].erase(v);
               yr[r[v].second].erase(v);
   
               r[v].first = x;
               r[v].second = y;
           }
           xr[r[v].first].insert(v);
           yr[r[v].second].insert(v);
   
           cout << r[v].first + 1 << ' ' << r[v].second + 1 << '\n';
   
           
       }
   
   }
   
   //5 5 5 3
   //3 4
   //4 2
   //1 1
   //3 1
   //2 1
   //2 L
   //3 D
   //1 L
   ```
